Reports Mockito method calls that has no argument passed. There are a couple of methods in Mockito that must be passed at least one argument (usually these methods take varargs as argument). 
@Test
void testMethod() {
    AnObject mock = Mockito.mock(AnObject.class, Mockito.withSettings().extraInterfaces());
    Mockito.verifyNoInteractions();
    Mockito.verifyNoMoreInteractions();
    Mockito.verifyZeroInteractions();
    Mockito.reset();
    Mockito.ignoreStubs();
    Mockito.inOrder();
}
